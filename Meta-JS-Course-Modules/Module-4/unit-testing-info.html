// async.js
function fetchData() {
  return new Promise(resolve => {
    setTimeout(() => resolve('data'), 100);
  });
}

function fetchDataWithError() {
  return new Promise((resolve, reject) => {
    setTimeout(() => reject(new Error('Network error')), 100);
  });
}

// async.test.js

describe('Async Functions', () => {
  // Method 1: async/await
  test('fetches data successfully', async () => {
    const data = await fetchData();
    expect(data).toBe('data');
  });

  // Method 2: returning Promise
  test('fetches data with promise', () => {
    return fetchData().then(data => {
      expect(data).toBe('data');
    });
  });

  // Testing promise rejection
  test('handles fetch error', async () => {
    await expect(fetchDataWithError()).rejects.toThrow('Network error');
  });
});

Setup & Lifecycle Hooks

Jest provides lifecycle hooks to set up and tear down test environments. These hooks help you prepare data, initialize objects, and clean up after tests run.
ü™ù Jest Lifecycle Hooks 
// database.js - Example service
class Database {
  constructor() {
    this.connected = false;
    this.users = [];
  }

  connect() {
    this.connected = true;
    return Promise.resolve();
  }

  disconnect() {
    this.connected = false;
    this.users = [];
  }

  addUser(user) {
    if (!this.connected) throw new Error('Not connected');
    this.users.push(user);
  }

  getUsers() {
    return this.users;
  }
}

module.exports = Database;

// database.test.js
const Database = require('./database');

describe('Database', () => {
  let db;

  // Runs before all tests in this describe block
  beforeAll(async () => {
    console.log('Setting up test database');
    db = new Database();
    await db.connect();
  });

  // Runs after all tests in this describe block
  afterAll(() => {
    console.log('Tearing down test database');
    db.disconnect();
  });

  // Runs before each test
  beforeEach(() => {
    db.users = []; // Reset users before each test
  });

  // Runs after each test
  afterEach(() => {
    console.log('Test completed');
  });

  test('should add user to database', () => {
    db.addUser({ id: 1, name: 'John' });
    expect(db.getUsers()).toHaveLength(1);
  });

  test('should start with empty users', () => {
    expect(db.getUsers()).toHaveLength(0);
  });
});

Jest Best Practices

Following best practices makes your tests more reliable, maintainable, and valuable. Here are key principles for writing effective Jest tests.

üìù Write Clear Test Names

Use descriptive test names that explain what is being tested and what the expected outcome is.

üéØ Test One Thing at a Time

Each test should focus on a single piece of functionality. This makes failures easier to diagnose.

üîÑ Follow AAA Pattern

Arrange (setup), Act (execute), Assert (verify). This structure makes tests easy to read and understand.

üö´ Avoid Test Dependencies

Tests should be independent and not rely on the order of execution or state from other tests.

üéØ Test Structure Best Practices 
// ‚ùå Bad: Vague test name, testing multiple things
test('user test', () => {
  const user = new User('John', 'john@email.com');
  expect(user.name).toBe('John');
  expect(user.email).toBe('john@email.com');
  expect(user.isValid()).toBe(true);
});

// ‚úÖ Good: Clear names, single responsibility, AAA pattern
describe('User', () => {
  test('should create user with valid name and email', () => {
    // Arrange
    const name = 'John';
    const email = 'john@email.com';

    // Act
    const user = new User(name, email);

    // Assert
    expect(user.name).toBe(name);
    expect(user.email).toBe(email);
  });

  test('should validate user with correct format', () => {
    // Arrange
    const user = new User('John', 'john@email.com');

    // Act
    const isValid = user.isValid();

    // Assert
    expect(isValid).toBe(true);
  });
});

‚ö†Ô∏è Common Testing Pitfalls

  • Testing implementation details: Focus on behavior, not how it's implemented
  • Brittle tests: Tests that break when code changes but behavior stays the same
  • Not testing edge cases: Always test boundary conditions and error scenarios
  • Slow tests: Avoid unnecessary async operations or complex setup

üé≠ Advanced Mocking Techniques 
// advancedMocking.test.js
describe('Advanced Mocking', () => {
  test('mock function calls and return values', () => {
    const mockFn = jest.fn();
    
    // Set return value
    mockFn.mockReturnValue(42);
    expect(mockFn()).toBe(42);
    
    // Check if function was called
    expect(mockFn).toHaveBeenCalled();
    expect(mockFn).toHaveBeenCalledTimes(1);
  });

  test('mock different return values for multiple calls', () => {
    const mockFn = jest.fn();
    
    mockFn
      .mockReturnValueOnce('first call')
      .mockReturnValueOnce('second call')
      .mockReturnValue('default');
    
    expect(mockFn()).toBe('first call');
    expect(mockFn()).toBe('second call');
    expect(mockFn()).toBe('default');
  });

  test('spy on object methods', () => {
    const calculator = {
      add: (a, b) => a + b,
      multiply: (a, b) => a * b
    };

    // Spy on method without changing its implementation
    const addSpy = jest.spyOn(calculator, 'add');
    
    const result = calculator.add(2, 3);
    
    expect(result).toBe(5);
    expect(addSpy).toHaveBeenCalledWith(2, 3);
    
    // Restore original implementation
    addSpy.mockRestore();
  });
});

// setup/customMatchers.js
expect.extend({
  toBeWithinRange(received, floor, ceiling) {
    const pass = received >= floor && received <= ceiling;
    if (pass) {
      return {
        message: () => `expected ${received} not to be within range ${floor} - ${ceiling}`,
        pass: true
      };
    } else {
      return {
        message: () => `expected ${received} to be within range ${floor} - ${ceiling}`,
        pass: false
      };
    }
  },

  toBeValidEmail(received) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    const pass = emailRegex.test(received);
    
    return {
      message: () => `expected ${received} ${pass ? 'not ' : ''}to be a valid email`,
      pass
    };
  }
});

// customMatchers.test.js
test('custom matchers work correctly', () => {
  expect(100).toBeWithinRange(90, 110);
  expect('test@example.com''adds 1 + 2 to equal 3', () => {
    expect(add(1, 2)).toBe(3);
  });

  test('multiplies 3 * 4 to equal 12', () => {
    expect(multiply(3, 4)).toBe(12);
  });
});

class Calculator {

const Calculator = require('./calculator');

  });

  test('should throw error when dividing by zero', () => {
    expect(() => calc.divide(10, 0)).toThrow('Division by zero');
  });

// user.js - Without tests, bugs can hide
function formatUserName(user) {
  return `${user.firstName} ${user.lastName}`; // Bug: no null check!
}

// user.test.js - Tests reveal edge cases
describe('formatUserName', () => {
  test('formats complete user name', () => {
    const user = { firstName: 'John', lastName: 'Doe' };
    expect(formatUserName(user)).toBe('John Doe');
  });

  test('handles missing lastName', () => {
    const user = { firstName: 'John' };
    expect(formatUserName(user)).toBe('John undefined'); // Reveals bug!
  });

  test('handles null user', () => {
    expect(() => formatUserName(null)).toThrow(); // Another bug!
  });
});

// Fixed version
function formatUserName(user) {
  if (!user) return 'Unknown User';
  const first = user.firstName || 'Unknown';
  const last = user.lastName || 'User';
  return `${first} ${last}`;
}

  expect(isValidEmail(null)).toBe(false);

// Common Jest Matchers
describe('Jest Matchers Examples', () => {
  test('equality matchers', () => {
    expect(2 + 2).toBe(4);                    // Exact equality
    expect({name: 'John'}).toEqual({name: 'John'}); // Deep equality
  });

  test('truthiness matchers', () => {
    expect('Hello').toBeTruthy();
    expect('').toBeFalsy();
    expect(null).toBeNull();
    expect(undefined).toBeUndefined();
  });

  test('number matchers', () => {
    expect(2 + 2).toBeGreaterThan(3);
    expect(3.14).toBeCloseTo(3.1);
  });

  test('array and object matchers', () => {
    expect([1, 2, 3]).toContain(2);
    expect({name: 'John'}).toHaveProperty('name');
  });
